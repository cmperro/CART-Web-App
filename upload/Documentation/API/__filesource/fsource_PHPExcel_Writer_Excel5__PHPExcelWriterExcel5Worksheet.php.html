  581.                     // $url = $url;
  582.  
  583.                 } else {

  603.     }

  605.     /**
  606.      * Write a cell range address in BIFF8
  607.      * always fixed range
  608.      * See section 2.5.14 in OpenOffice.org's Documentation of the Microsoft Excel File Format
  609.      *
  610.      * @param string $range E.g. 'A1' or 'A1:B6'
  611.      * @return string Binary data
  612.      */
  613.     private function _writeBIFF8CellRangeAddressFixed($range = 'A1')
  614.     {
  615.         $explodes = explode(':', $range);
  616.  
  617.         // extract first cell, e.g. 'A1'
  618.         $firstCell = $explodes[0];
  619.  
  620.         // extract last cell, e.g. 'B6'
  621.         if (count($explodes) == 1) {
  622.             $lastCell = $firstCell;
  623.         } else {
  624.             $lastCell = $explodes[1];
  625.         }
  626.  
  627.         $firstCellCoordinates = PHPExcel_Cell::coordinateFromString($firstCell); // e.g. array(0, 1)
  628.         $lastCellCoordinates  = PHPExcel_Cell::coordinateFromString($lastCell);  // e.g. array(1, 6)
  629.  
  630.         $data = pack('vvvv',
  631.             $firstCellCoordinates[1] - 1,
  632.             $lastCellCoordinates[1] - 1,
  633.             PHPExcel_Cell::columnIndexFromString($firstCellCoordinates[0]) - 1,
  634.             PHPExcel_Cell::columnIndexFromString($lastCellCoordinates[0]) - 1
  635.         );
  636.  
  637.         return $data;
  638.     }

  640.     /**
  641.      * Retrieves data from memory in one chunk, or from disk in $buffer

  669.     }

  743.     /**
  744.      * Write a double to the specified row and column (zero indexed).
  745.      * An integer can be written as a double. Excel will display an
  746.      * integer. $format is optional.
  747.      *
  748.      * Returns  0 : normal termination
  749.      *         -2 : row or column out of range
  750.      *
  751.      * @param integer $row    Zero indexed row
  752.      * @param integer $col    Zero indexed column
  753.      * @param float   $num    The number to write
  754.      * @param mixed   $format The optional XF format
  755.      * @return integer 
  756.      */
  757.     private function _writeNumber($row, $col, $num, $xfIndex)
  758.     {
  759.         $record    = 0x0203;                 // Record identifier
  760.         $length    = 0x000E;                 // Number of bytes to follow
  761.  
  762.         $header    = pack("vv",  $record, $length);
  763.         $data      = pack("vvv", $row, $col, $xfIndex);
  764.         $xl_double = pack("d",   $num);
  765.         if (PHPExcel_Writer_Excel5_BIFFwriter::getByteOrder()) { // if it's Big Endian
  766.             $xl_double = strrev($xl_double);
  767.         }
  768.  
  769.         $this->_append($header.$data.$xl_double);
  770.         return(0);
  771.     }
  772.  
  773.     /**
  774.      * Write a LABELSST record or a LABEL record. Which one depends on BIFF version
  775.      *
  776.      * @param int $row Row index (0-based)
  777.      * @param int $col Column index (0-based)
  778.      * @param string $str The string
  779.      * @param int $xfIndex Index to XF record
  780.      */
  781.     private function _writeString($row, $col, $str, $xfIndex)
  782.     {

  789.     /**
  790.      * Write a string to the specified row and column (zero indexed).
  791.      * NOTE: there is an Excel 5 defined limit of 255 characters.
  792.      * $format is optional.
  793.      * Returns  0 : normal termination
  794.      *         -2 : row or column out of range
  795.      *         -3 : long string truncated to 255 chars
  796.      *
  797.      * @access public
  798.      * @param integer $row    Zero indexed row
  799.      * @param integer $col    Zero indexed column
  800.      * @param string  $str    The string to write
  801.      * @param mixed   $format The XF format for the cell
  802.      * @return integer 
  803.      */
  804.     private function _writeLabel($row, $col, $str, $xfIndex)
  805.     {
  806.         $strlen    = strlen($str);
  807.         $record    = 0x0204;                   // Record identifier
  808.         $length    = 0x0008 + $strlen;         // Bytes to follow
  809.  
  810.         $str_error = 0;
  811.  
  812.         if ($strlen > $this->_xls_strmax) { // LABEL must be < 255 chars
  813.             $str       = substr($str, 0, $this->_xls_strmax);
  814.             $length    = 0x0008 + $this->_xls_strmax;
  815.             $strlen    = $this->_xls_strmax;
  816.             $str_error = -3;
  817.         }
  818.  
  819.         $header    = pack("vv",   $record, $length);
  820.         $data      = pack("vvvv", $row, $col, $xfIndex, $strlen);
  821.         $this->_append($header . $data . $str);
  822.         return($str_error);
  823.     }
  824.  
  825.     /**
  826.      * Write a string to the specified row and column (zero indexed).
  827.      * This is the BIFF8 version (no 255 chars limit).
  828.      * $format is optional.
  829.      * Returns  0 : normal termination
  830.      *         -2 : row or column out of range
  831.      *         -3 : long string truncated to 255 chars
  832.      *
  833.      * @access public
  834.      * @param integer $row    Zero indexed row
  835.      * @param integer $col    Zero indexed column
  836.      * @param string  $str    The string to write
  837.      * @param mixed   $format The XF format for the cell
  838.      * @return integer 
  839.      */
  840.     private function _writeLabelSst($row, $col, $str, $xfIndex)
  841.     {
  842.         $record    = 0x00FD;                   // Record identifier
  843.         $length    = 0x000A;                   // Bytes to follow
  844.  
  845.         $str = PHPExcel_Shared_String::UTF8toBIFF8UnicodeLong($str);
  846.  
  847.         /* check if string is already present */
  848.         if (!isset($this->_str_table[$str])) {
  849.             $this->_str_table[$str] = $this->_str_unique++;
  850.         }
  851.         $this->_str_total++;
  852.  
  853.         $header    = pack('vv',   $record, $length);
  854.         $data      = pack('vvvV', $row, $col, $xfIndex, $this->_str_table[$str]);
  855.         $this->_append($header.$data);
  856.     }
  857.  
  858.     /**
  859.      * Writes a note associated with the cell given by the row and column.
  860.      * NOTE records don't have a length limit.
  861.      *
  862.      * @param integer $row    Zero indexed row
  863.      * @param integer $col    Zero indexed column
  864.      * @param string  $note   The note to write
  865.      */
  866.     private function _writeNote($row, $col, $note)
  867.     {
  868.         $note_length    = strlen($note);
  869.         $record         = 0x001C;                // Record identifier
  870.         $max_length     = 2048;                  // Maximun length for a NOTE record
  871.         //$length      = 0x0006 + $note_length;    // Bytes to follow
  872.  
  873.         // Length for this record is no more than 2048 + 6
  874.         $length    = 0x0006 + min($note_length, 2048);
  875.         $header    = pack("vv",   $record, $length);
  876.         $data      = pack("vvv", $row, $col, $note_length);
  877.         $this->_append($header . $data . substr($note, 0, 2048));
  878.  
  879.         for ($i = $max_length; $i < $note_length; $i += $max_length) {
  880.             $chunk  = substr($note, $i, $max_length);
  881.             $length = 0x0006 + strlen($chunk);
  882.             $header = pack("vv",   $record, $length);
  883.             $data   = pack("vvv", -1, 0, strlen($chunk));

  936.     /**
  937.      * Write a formula to the specified row and column (zero indexed).
  938.      * The textual representation of the formula is passed to the parser in
  939.      * Parser.php which returns a packed binary string.
  940.      *
  941.      * Returns  0 : normal termination
  942.      *         -1 : formula errors (bad formula)
  943.      *         -2 : row or column out of range
  944.      *
  945.      * @param integer $row     Zero indexed row
  946.      * @param integer $col     Zero indexed column
  947.      * @param string  $formula The formula text string
  948.      * @param mixed   $format  The optional XF format
  949.      * @param mixed   $calculatedValue  Calculated value
  950.      * @return integer 
  951.      */
  952.     private function _writeFormula($row, $col, $formula, $xfIndex, $calculatedValue)
  953.     {
  954.         $record    = 0x0006;     // Record identifier
  955.  
  956.         // Initialize possible additional value for STRING record that should be written after the FORMULA record?
  957.         $stringValue = null;
  958.  
  959.         // calculated value
  960.         if (isset($calculatedValue)) {
  961.  
  962.             // Since we can't yet get the data type of the calculated value,
  963.             // we use best effort to determine data type
  964.  
  965.             if (is_bool($calculatedValue)) {
  966.                 // Boolean value
  967.                 $num = pack('CCCvCv', 0x01, 0x00, (int)$calculatedValue, 0x00, 0x00, 0xFFFF);
  968.  
  969.             } elseif (is_int($calculatedValue) || is_float($calculatedValue)) {
  970.                 // Numeric value
  971.                 $num = pack('d', $calculatedValue);
  972.  
  973.             } elseif (is_string($calculatedValue)) {
  974.                 if (array_key_exists($calculatedValue, PHPExcel_Cell_DataType::getErrorCodes())) {
  975.                     // Error value
  976.                     $num = pack('CCCvCv', 0x02, 0x00, $this->_mapErrorCode($calculatedValue), 0x00, 0x00, 0xFFFF);
  977.  
  978.                 } elseif ($calculatedValue === '' && $this->_BIFF_version == 0x0600) {
  979.                     // Empty string (and BIFF8)
  980.                     $num = pack('CCCvCv', 0x03, 0x00, 0x00, 0x00, 0x00, 0xFFFF);
  981.  
  982.                 } else {
  983.                     // Non-empty string value (or empty string BIFF5)
  984.                     $stringValue = $calculatedValue;
  985.                     $num = pack('CCCvCv', 0x00, 0x00, 0x00, 0x00, 0x00, 0xFFFF);
  986.  
  987.                 }
  988.  
  989.             } else {
  990.                 // We are really not supposed to reach here
  991.                 $num = pack('d', 0x00);
  992.  
  993.             }
  994.  
  995.         } else {
  996.             $num = pack('d', 0x00);
  997.         }
  998.  
  999.         $grbit     = 0x03;                // Option flags
  1000.         $unknown   = 0x0000;              // Must be zero
  1001.  
  1002.         // Strip the '=' or '@' sign at the beginning of the formula string
  1003.         if (preg_match("/^=/", $formula)) {
  1004.             $formula = preg_replace("/(^=)/", "", $formula);
  1005.         } elseif (preg_match("/^@/", $formula)) {
  1006.             $formula = preg_replace("/(^@)/", "", $formula);
  1007.         } else {
  1008.             // Error handling
  1009.             $this->_writeString($row, $col, 'Unrecognised character for formula');
  1010.             return -1;
  1011.         }
  1012.  
  1013.         // Parse the formula using the parser in Parser.php
  1014.         try {
  1015.             $error = $this->_parser->parse($formula);
  1016.             $formula = $this->_parser->toReversePolish();
  1017.  
  1018.             $formlen    = strlen($formula);    // Length of the binary string
  1019.             $length     = 0x16 + $formlen;     // Length of the record data
  1020.  
  1021.             $header    = pack("vv",      $record, $length);
  1022.  
  1023.             $data      = pack("vvv", $row, $col, $xfIndex)
  1024.                         . $num
  1025.                         . pack("vVv", $grbit, $unknown, $formlen);
  1026.             $this->_append($header . $data . $formula);
  1027.  
  1028.             // Append also a STRING record if necessary
  1029.             if ($stringValue !== null) {
  1030.                 $this->_writeStringRecord($stringValue);
  1031.             }
  1032.  
  1033.             return 0;
  1034.  
  1035.         } catch (Exception $e) {
  1036.             // do nothing
  1037.         }
  1038.  
  1039.     }
  1040.  
  1041.     /**
  1042.      * Write a STRING record. This
  1043.      *
  1044.      * @param string $stringValue 
  1045.      */
  1046.     private function _writeStringRecord($stringValue)
  1047.     {
  1048.         $record = 0x0207;     // Record identifier
  1049.         $data = PHPExcel_Shared_String::UTF8toBIFF8UnicodeLong($stringValue);

  1052.         $header = pack('vv', $record, $length);
  1053.  
  1054.         $this->_append($header . $data);
  1055.     }
  1056.  
  1057.     /**
  1058.      * Write a hyperlink.

  1060.      * the invisible link. The visible label is the same as the link unless an

  1081.     }
  1082.  
  1083.     /**

  1085.      * written to a range of cells. This function also decides the type of hyperlink
  1086.      * to be written. These are either, Web (http, ftp, mailto), Internal
  1087.      * (Sheet1!A1) or external ('c:\temp\foo.xls#Sheet1!A1').

  1091.      * @param integer $row1   Start row

  1093.      * @param integer $row2   End row
  1094.      * @param integer $col2   End column
  1095.      * @param string  $url    URL string
  1096.      * @return integer 
  1097.      */
  1098.  
  1099.     function _writeUrlRange($row1, $col1, $row2, $col2, $url)
  1100.     {
  1101.  
  1102.         // Check for internal/external sheet links or default to web link

  1110.     }
  1111.  
  1112.  
  1113.     /**
  1114.      * Used to write http, ftp and mailto hyperlinks.
  1115.      * The link type ($options) is 0x03 is the same as absolute dir ref without
  1116.      * sheet. However it is differentiated by the $unknown2 data stream.
  1117.      *
  1118.      * @access private

  1121.      * @param integer $col1   Start column
  1122.      * @param integer $row2   End row
  1123.      * @param integer $col2   End column
  1124.      * @param string  $url    URL string
  1125.      * @return integer 
  1126.      */
  1127.     function _writeUrlWeb($row1, $col1, $row2, $col2, $url)
  1128.     {
  1129.         $record      = 0x01B8;                       // Record identifier
  1130.         $length      = 0x00000;                      // Bytes to follow
  1131.  
  1132.         // Pack the undocumented parts of the hyperlink stream
  1133.         $unknown1    = pack("H*", "D0C9EA79F9BACE118C8200AA004BA90B02000000");

  1152.  
  1153.         // Write the packed data
  1154.         $this->_append($header . $data .

  1156.                        $unknown2 . $url_len . $url);
  1157.         return 0;
  1158.     }
  1159.  
  1160.     /**

  1163.      * @access private

  1165.      * @param integer $row1   Start row

  1168.      * @param integer $col2   End column

  1208.  
  1209.     /**
  1210.      * Write links to external directory names such as 'c:\foo.xls',
  1211.      * c:\foo.xls#Sheet1!A1', '../../foo.xls'. and '../../foo.xls#Sheet1!A1'.
  1212.      *
  1213.      * Note: Excel writes some relative links with the $dir_long string. We ignore
  1214.      * these cases for the sake of simpler code.
  1215.      *
  1216.      * @access private
  1217.      * @see _writeUrl()
  1218.      * @param integer $row1   Start row
  1219.      * @param integer $col1   Start column
  1220.      * @param integer $row2   End row
  1221.      * @param integer $col2   End column
  1222.      * @param string  $url    URL string
  1223.      * @return integer 
  1224.      */
  1225.     function _writeUrlExternal($row1, $col1, $row2, $col2, $url)
  1226.     {
  1227.         // Network drives are different. We will handle them separately
  1228.         // MS/Novell network drives and shares start with \\
  1229.         if (preg_match('[^external:\\\\]', $url)) {
  1230.             return; //($this->_writeUrlExternal_net($row1, $col1, $row2, $col2, $url, $str, $format));
  1231.         }
  1232.  
  1233.         $record      = 0x01B8;                       // Record identifier
  1234.         $length      = 0x00000;                      // Bytes to follow
  1235.  
  1236.         // Strip URL type and change Unix dir separator to Dos style (if needed)
  1237.         //
  1238.         $url = preg_replace('/^external:/', '', $url);
  1239.         $url = preg_replace('/\//', "\\", $url);
  1240.  
  1241.         // Determine if the link is relative or absolute:
  1242.         //   relative if link contains no dir separator, "somefile.xls"
  1243.         //   relative if link starts with up-dir, "..\..\somefile.xls"
  1244.         //   otherwise, absolute
  1245.  
  1246.         $absolute = 0x00; // relative path
  1247.         if ( preg_match('/^[A-Z]:/', $url) ) {
  1248.             $absolute = 0x02; // absolute path on Windows, e.g. C:\...
  1249.         }
  1250.         $link_type               = 0x01 | $absolute;
  1251.  
  1252.         // Determine if the link contains a sheet reference and change some of the
  1253.         // parameters accordingly.
  1254.         // Split the dir name and sheet name (if it exists)
  1255.         $dir_long = $url;
  1256.         if (preg_match("/\#/", $url)) {
  1257.             $link_type |= 0x08;
  1258.         }
  1259.  
  1260.  
  1261.         // Pack the link type
  1262.         $link_type   = pack("V", $link_type);
  1263.

  2577.     /**
  2578.      * Calculate the vertices that define the position of the image as required by
  2579.      * the OBJ record.
  2580.      *
  2581.      *         +------------+------------+
  2582.      *         |     A      |      B     |
  2583.      *   +-----+------------+------------+
  2584.      *   |     |(x1,y1)     |            |

  2586.      *   |     |    |              |     |
  2587.      *   |     |    |              |     |
  2588.      *   +-----+----|    BITMAP    |-----+
  2589.      *   |     |    |              |     |
  2590.      *   |  2  |    |______________.     |
  2591.      *   |     |            |        (B2)|
  2592.      *   |     |            |     (x2,y2)|
  2593.      *   +---- +------------+------------+
  2594.      *
  2595.      * Example of a bitmap that covers some of the area from cell A1 to cell B2.
  2596.      *
  2597.      * Based on the width and height of the bitmap we need to calculate 8 vars:
  2598.      *     $col_start, $row_start, $col_end, $row_end, $x1, $y1, $x2, $y2.
  2599.      * The width and height of the cells are also variable and have to be taken into
  2600.      * account.
  2601.      * The values of $col_start and $row_start are passed in from the calling
  2602.      * function. The values of $col_end and $row_end are calculated by subtracting
  2603.      * the width and height of the bitmap from the width and height of the
  2604.      * underlying cells.
  2605.      * The vertices are expressed as a percentage of the underlying cell width as
  2606.      * follows (rhs values are in pixels):
  2607.      *
  2608.      *       x1 = X / W *1024
  2609.      *       y1 = Y / H *256

  2611.      *       y2 = (Y-1) / H *256
  2612.      *
  2613.      *       Where:  X is distance from the left side of the underlying cell
  2614.      *               Y is distance from the top of the underlying cell
  2615.      *               W is the width of the cell
  2616.      *               H is the height of the cell
  2617.      * The SDK incorrectly states that the height should be expressed as a
  2618.      *        percentage of 1024.
  2619.      *
  2620.      * @access private
  2621.      * @param integer $col_start Col containing upper left corner of object
  2622.      * @param integer $row_start Row containing top left corner of object
  2623.      * @param integer $x1        Distance to left side of object
  2624.      * @param integer $y1        Distance to top of object
  2625.      * @param integer $width     Width of image frame
  2626.      * @param integer $height    Height of image frame
  2627.      */
  2628.     function _positionImage($col_start, $row_start, $x1, $y1, $width, $height)
  2629.     {
  2630.         // Initialise end cell to the same as the start cell

  2632.         $row_end    = $row_start;  // Row containing bottom right corner of object
  2633.

  2637.         }

  2644.

  2655.         }
  2656.  
  2657.         // Bitmap isn't allowed to start or finish in a hidden cell, i.e. a cell
  2658.         // with zero eight or width.
  2659.         //
  2660.         if (PHPExcel_Shared_Excel5::sizeCol($this->_phpSheet, PHPExcel_Cell::stringFromColumnIndex($col_start)) == 0) {
  2661.             return;
  2662.         }
  2663.         if (PHPExcel_Shared_Excel5::sizeCol($this->_phpSheet, PHPExcel_Cell::stringFromColumnIndex($col_end))   == 0) {
  2664.             return;
  2665.         }

  2674.         $x1 = $x1     / PHPExcel_Shared_Excel5::sizeCol($this->_phpSheet, PHPExcel_Cell::stringFromColumnIndex($col_start))   * 1024;
  2675.         $y1 = $y1     / PHPExcel_Shared_Excel5::sizeRow($this->_phpSheet, $row_start + 1)   *  256;

  2677.         $y2 = $height / PHPExcel_Shared_Excel5::sizeRow($this->_phpSheet, $row_end + 1)     *  256; // Distance to bottom of object
  2678.  
  2679.         $this->_storeObjPicture($col_start, $x1,
  2680.                                  $row_start, $y1,
  2681.                                  $col_end, $x2,
  2682.                                  $row_end, $y2);
  2683.     }

  2771.      */
  2772.     function _processBitmapGd($image) {
  2773.         $width = imagesx($image);
  2774.         $height = imagesy($image);
  2775.  
  2776.         $data = pack("Vvvvv", 0x000c, $width, $height, 0x01, 0x18);
  2777.         for ($j=$height; $j--; ) {
  2778.             for ($i=0; $i < $width; ++$i) {
  2779.                 $color = imagecolorsforindex($image, imagecolorat($image, $i, $j));
  2780.                 foreach (array("red", "green", "blue") as $key) {
  2781.                     $color[$key] = $color[$key] + round((255 - $color[$key]) * $color["alpha"] / 127);
  2782.                 }
  2783.                 $data .= chr($color["blue"]) . chr($color["green"]) . chr($color["red"]);
  2784.             }
  2785.             if (3*$width % 4) {
  2786.                 $data .= str_repeat("\x00", 4 - 3*$width % 4);
  2787.             }
  2788.         }
  2789.  
  2790.         return array($width, $height, strlen($data), $data);
  2791.     }
