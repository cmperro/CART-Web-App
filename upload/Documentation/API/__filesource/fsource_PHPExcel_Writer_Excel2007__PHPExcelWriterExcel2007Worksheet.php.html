  351.     /**

  358.     private function _writeCols(PHPExcel_Shared_XMLWriter $objWriter = null, PHPExcel_Worksheet $pSheet = null)
  359.     {

  375.                         $objWriter->writeAttribute('width',        '9.10');
  376.                     } else {
  377.                         // Width set
  378.                         $objWriter->writeAttribute('width',        PHPExcel_Shared_String::FormatNumber($colDimension->getWidth()));
  379.                     }
  380.  
  381.                     // Column visibility

  384.                     }
  385.  
  386.                     // Auto size?

  389.                     }
  390.  
  391.                     // Custom width?
  392.                     if ($colDimension->getWidth() != $pSheet->getDefaultColumnDimension()->getWidth()) {
  393.                         $objWriter->writeAttribute('customWidth',    'true');
  394.                     }
  395.  
  396.                     // Collapsed
  397.                     if ($colDimension->getCollapsed() == true) {
  398.                         $objWriter->writeAttribute('collapsed',        'true');
  399.                     }
  400.  
  401.                     // Outline level
  402.                     if ($colDimension->getOutlineLevel() > 0) {
  403.                         $objWriter->writeAttribute('outlineLevel',    $colDimension->getOutlineLevel());
  404.                     }
  405.  
  406.                     // Style
  407.                     $objWriter->writeAttribute('style', $colDimension->getXfIndex());
  408.  
  409.                     $objWriter->endElement();
  410.                 }
  411.  
  412.             $objWriter->endElement();
  413.         }

  501.                              $objWriter->writeElement('formula',    'LEFT(' . $cellCoordinate . ',' . strlen($conditional->getText()) . ')="' . $conditional->getText() . '"');
  502.                          } else if ($conditional->getConditionType() == PHPExcel_Style_Conditional::CONDITION_CONTAINSTEXT
  503.                              && $conditional->getOperatorType() == PHPExcel_Style_Conditional::OPERATOR_ENDSWITH
  504.                              && !is_null($conditional->getText())) {
  505.                              $objWriter->writeElement('formula',    'RIGHT(' . $cellCoordinate . ',' . strlen($conditional->getText()) . ')="' . $conditional->getText() . '"');
  506.                          } else if ($conditional->getConditionType() == PHPExcel_Style_Conditional::CONDITION_CONTAINSTEXT
  507.                              && $conditional->getOperatorType() == PHPExcel_Style_Conditional::OPERATOR_NOTCONTAINS
  508.                              && !is_null($conditional->getText())) {
  509.                              $objWriter->writeElement('formula',    'ISERROR(SEARCH("' . $conditional->getText() . '",' . $cellCoordinate . '))');
  510.                          } else if ($conditional->getConditionType() == PHPExcel_Style_Conditional::CONDITION_CELLIS
  511.                             || $conditional->getConditionType() == PHPExcel_Style_Conditional::CONDITION_CONTAINSTEXT
  512.                             || $conditional->getConditionType() == PHPExcel_Style_Conditional::CONDITION_EXPRESSION) {

  516.                             }
  517.                         }
  518.  
  519.                         $objWriter->endElement();
