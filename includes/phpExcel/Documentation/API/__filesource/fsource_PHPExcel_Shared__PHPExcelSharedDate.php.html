  95.     public static function ExcelToPHP($dateValue = 0) {
  96.         if (self::$ExcelBaseDate == self::CALENDAR_WINDOWS_1900) {
  97.             $myExcelBaseDate = 25569;
  98.             //    Adjust for the spurious 29-Feb-1900 (Day 60)
  99.             if ($dateValue < 60) {
  100.                 --$myExcelBaseDate;
  101.             }
  102.         } else {
  103.             $myExcelBaseDate = 24107;
  104.         }
  105.  
  106.         // Perform conversion
  107.         if ($dateValue >= 1) {
  108.             $utcDays = $dateValue - $myExcelBaseDate;
  109.             $returnValue = round($utcDays * 24 * 60 * 60);
  110.             if (($returnValue <= PHP_INT_MAX) && ($returnValue >= -PHP_INT_MAX)) {
  111.                 $returnValue = (integer) $returnValue;
  112.             }
  113.         } else {
  114.             $hours = round($dateValue * 24);
  115.             $mins = round($dateValue * 24 * 60) - round($hours * 60);
  116.             $secs = round($dateValue * 24 * 60 * 60) - round($hours * 60 * 60) - round($mins * 60);
  117.             $returnValue = (integer) gmmktime($hours, $mins, $secs);
  118.         }
  119.  
  120.         // Return
  121.         return $returnValue;
  122.     }    //    function ExcelToPHP()

  182.      */
  183.     public static function FormattedPHPToExcel($year, $month, $day, $hours=0, $minutes=0, $seconds=0) {
  184.         if (self::$ExcelBaseDate == self::CALENDAR_WINDOWS_1900) {
  185.             //
  186.             //    Fudge factor for the erroneous fact that the year 1900 is treated as a Leap Year in MS Excel
  187.             //    This affects every date following 28th February 1900
  188.             //
  189.             $excel1900isLeapYear = True;
  190.             if (($year == 1900) && ($month <= 2)) { $excel1900isLeapYear = False; }
  191.             $myExcelBaseDate = 2415020;
  192.         } else {
  193.             $myExcelBaseDate = 2416481;
  194.             $excel1900isLeapYear = False;
  195.         }
  196.  
  197.         //    Julian base date Adjustment
  198.         if ($month > 2) {
  199.             $month = $month - 3;
  200.         } else {
  201.             $month = $month + 9;
  202.             --$year;
  203.         }
  204.  
  205.         //    Calculate the Julian Date, then subtract the Excel base date (JD 2415020 = 31-Dec-1899 Giving Excel Date of 0)
  206.         $century = substr($year,0,2);
  207.         $decade = substr($year,2,2);
  208.         $excelDate = floor((146097 * $century) / 4) + floor((1461 * $decade) / 4) + floor((153 * $month + 2) / 5) + $day + 1721119 - $myExcelBaseDate + $excel1900isLeapYear;
  209.  
  210.         $excelTime = (($hours * 3600) + ($minutes * 60) + $seconds) / 86400;
  211.  
  212.         return (float) $excelDate + $excelTime;
  213.     }    //    function FormattedPHPToExcel()
