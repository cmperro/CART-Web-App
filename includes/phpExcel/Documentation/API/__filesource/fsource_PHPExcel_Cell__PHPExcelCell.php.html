  465.     public function isInRange($pRange = 'A1:A1')

  469.

  487.  
  488.            // Translate properties

  490.         $myRow            = $this->getRow();
  491.  
  492.         // Verify if cell is in range
  493.         return (
  494.                 ($rangeStart[0] <= $myColumn && $rangeEnd[0] >= $myColumn) &&
  495.                 ($rangeStart[1] <= $myRow && $rangeEnd[1] >= $myRow)
  496.         );
