  380.         // Parse formula into a tree of tokens
  381.         $tokenisedFormula = PHPExcel_Calculation::getInstance()->parseFormula($pFormula);
  382.  
  383.         $newCellTokens = $cellTokens = array();
  384.         $adjustCount = 0;
  385.         //    Build the translation table of cell tokens
  386.         foreach($tokenisedFormula as $token) {
  387.             $token = $token['value'];
  388.             if (preg_match('/^'.PHPExcel_Calculation::CALCULATION_REGEXP_CELLREF.'$/i', $token, $matches)) {
  389.                 list($column,$row) = PHPExcel_Cell::coordinateFromString($token);
  390.                 //    Max worksheet size is 1,048,576 rows by 16,384 columns in Excel 2007, so our adjustments need to be at least one digit more
  391.                 $column = PHPExcel_Cell::columnIndexFromString($column) + 100000;
  392.                 $row += 10000000;
  393.                 $cellIndex = $column.$row;
  394.                 if (!isset($cellTokens[$cellIndex])) {
  395.                     $newReference = $this->updateCellReference($token, $pBefore, $pNumCols, $pNumRows);
  396.                     if ($newReference !== $token) {
  397.                         $newCellTokens[$cellIndex] = preg_quote($newReference);
  398.                         $cellTokens[$cellIndex] = '/(?<![A-Z])'.preg_quote($token).'(?!\d)/i';
  399.                         ++$adjustCount;
  400.                     }
  401.                 }
  402.             }
  403.         }
  404.         if ($adjustCount == 0) {
  405.             return $pFormula;
  406.         }
  407.         krsort($cellTokens);
  408.         krsort($newCellTokens);
  409.  
  410.         //    Update cell references in the formula
  411.         $formulaBlocks = explode('"',$pFormula);
  412.         foreach($formulaBlocks as $i => &$formulaBlock) {
  413.             //    Only count/replace in alternate array entries
  414.             if (($i % 2) == 0) {
  415.                 $formulaBlock = preg_replace($cellTokens,$newCellTokens,$formulaBlock);
  416.             }
  417.         }
  418.         unset($formulaBlock);
  419.  
  420.         //    Then rebuild the formula string
  421.         return implode('"',$formulaBlocks);
  422.     }

  424.     /**
  425.      * Update cell reference
  426.      *
  427.      * @param    string    $pCellRange            Cell range
  428.      * @param    int        $pBefore            Insert before this one
  429.      * @param    int        $pNumCols            Number of columns to increment
  430.      * @param    int        $pNumRows            Number of rows to increment
  431.      * @return    string    Updated cell range
  432.      * @throws    Exception
  433.      */
  434.     public function updateCellReference($pCellRange = 'A1', $pBefore = 'A1', $pNumCols = 0, $pNumRows = 0) {
  435.         // Is it in another worksheet? Will not have to update anything.
  436.         if (strpos($pCellRange, "!") !== false) {
  437.             return $pCellRange;
  438.         // Is it a range or a single cell?
  439.         } elseif (strpos($pCellRange, ':') === false && strpos($pCellRange, ',') === false) {
  440.             // Single cell
  441.             return $this->_updateSingleCellReference($pCellRange, $pBefore, $pNumCols, $pNumRows);
  442.         } elseif (strpos($pCellRange, ':') !== false || strpos($pCellRange, ',') !== false) {
  443.             // Range
  444.             return $this->_updateCellRange($pCellRange, $pBefore, $pNumCols, $pNumRows);
  445.         } else {
  446.             // Return original
  447.             return $pCellRange;
  448.         }
  449.     }

  451.     /**
  452.      * Update named formulas (i.e. containing worksheet references / named ranges)
  453.      *
  454.      * @param PHPExcel $pPhpExcel    Object to update
  455.      * @param string $oldName        Old name (name to replace)
  456.      * @param string $newName        New name
  457.      */
  458.     public function updateNamedFormulas(PHPExcel $pPhpExcel, $oldName = '', $newName = '') {
  459.         if ($oldName == '') {
  460.             return;
  461.         }
  462.         
  463.         foreach ($pPhpExcel->getWorksheetIterator() as $sheet) {
  464.             foreach ($sheet->getCellCollection(false) as $cell) {
  465.                 if (!is_null($cell) && $cell->getDataType() == PHPExcel_Cell_DataType::TYPE_FORMULA) {
  466.                     $formula = $cell->getValue();
  467.                     if (strpos($formula, $oldName) !== false) {
  468.                         $formula = str_replace("'" . $oldName . "'!", "'" . $newName . "'!", $formula);
  469.                         $formula = str_replace($oldName . "!", $newName . "!", $formula);
  470.                         $cell->setValueExplicit($formula, PHPExcel_Cell_DataType::TYPE_FORMULA);
  471.                     }
  472.                 }
  473.             }
  474.         }
  475.     }

  477.     /**
  478.      * Update cell range
  479.      *
  480.      * @param    string    $pCellRange            Cell range
  481.      * @param    int        $pBefore            Insert before this one
  482.      * @param    int        $pNumCols            Number of columns to increment
  483.      * @param    int        $pNumRows            Number of rows to increment
  484.      * @return    string    Updated cell range
  485.      * @throws    Exception
  486.      */
  487.     private function _updateCellRange($pCellRange = 'A1:A1', $pBefore = 'A1', $pNumCols = 0, $pNumRows = 0) {
  488.         if (strpos($pCellRange,':') !== false || strpos($pCellRange, ',') !== false) {
  489.             // Update range
  490.             $range = PHPExcel_Cell::splitRange($pCellRange);
  491.             for ($i = 0; $i < count($range); ++$i) {
  492.                 for ($j = 0; $j < count($range[$i]); ++$j) {
  493.                     $range[$i][$j] = $this->_updateSingleCellReference($range[$i][$j], $pBefore, $pNumCols, $pNumRows);
  494.                 }
  495.             }
  496.  
  497.             // Recreate range string
  498.             return PHPExcel_Cell::buildRange($range);
  499.         } else {
  500.             throw new Exception("Only cell ranges may be passed to this method.");
  501.         }
  502.     }

  504.     /**
  505.      * Update single cell reference
  506.      *
  507.      * @param    string    $pCellReference        Single cell reference
  508.      * @param    int        $pBefore            Insert before this one
  509.      * @param    int        $pNumCols            Number of columns to increment
  510.      * @param    int        $pNumRows            Number of rows to increment
  511.      * @return    string    Updated cell reference
  512.      * @throws    Exception
  513.      */
  514.     private function _updateSingleCellReference($pCellReference = 'A1', $pBefore = 'A1', $pNumCols = 0, $pNumRows = 0) {
  515.         if (strpos($pCellReference, ':') === false && strpos($pCellReference, ',') === false) {
  516.             // Get coordinates of $pBefore
  517.             $beforeColumn    = 'A';
  518.             $beforeRow        = 1;
  519.             list($beforeColumn, $beforeRow) = PHPExcel_Cell::coordinateFromString( $pBefore );
  520.  
  521.             // Get coordinates
  522.             $newColumn    = 'A';
  523.             $newRow    = 1;
  524.             list($newColumn, $newRow) = PHPExcel_Cell::coordinateFromString( $pCellReference );
  525.  
  526.             // Make sure the reference can be used
  527.             if ($newColumn == '' && $newRow == '')
  528.             {
  529.                 return $pCellReference;
  530.             }
  531.  
  532.             // Verify which parts should be updated
  533.             $updateColumn = (PHPExcel_Cell::columnIndexFromString($newColumn) >= PHPExcel_Cell::columnIndexFromString($beforeColumn))
  534.                             && (strpos($newColumn, '$') === false)
  535.                             && (strpos($beforeColumn, '$') === false);
  536.  
  537.             $updateRow = ($newRow >= $beforeRow)
  538.                             && (strpos($newRow, '$') === false)
  539.                             && (strpos($beforeRow, '$') === false);
  540.  
  541.             // Create new column reference
  542.             if ($updateColumn) {
  543.                 $newColumn    = PHPExcel_Cell::stringFromColumnIndex( PHPExcel_Cell::columnIndexFromString($newColumn) - 1 + $pNumCols );
  544.             }
  545.  
  546.             // Create new row reference
  547.             if ($updateRow) {
  548.                 $newRow    = $newRow + $pNumRows;
  549.             }
  550.  
  551.             // Return new reference
  552.             return $newColumn . $newRow;
  553.         } else {
  554.             throw new Exception("Only single cell references may be passed to this method.");
  555.         }
  556.     }
