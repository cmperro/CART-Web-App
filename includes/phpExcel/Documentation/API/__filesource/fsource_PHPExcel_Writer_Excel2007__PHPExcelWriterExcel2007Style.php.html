  386.     {

  395.                 case PHPExcel_Style_Borders::DIAGONAL_DOWN:

  397.                     $objWriter->writeAttribute('diagonalDown',     'true');
  398.                     break;
  399.                 case PHPExcel_Style_Borders::DIAGONAL_BOTH:
  400.                     $objWriter->writeAttribute('diagonalUp',     'true');
  401.                     $objWriter->writeAttribute('diagonalDown',     'true');
  402.                     break;
  403.             }
  404.  
  405.             // BorderPr
  406.             $this->_writeBorderPr($objWriter, 'left',             $pBorders->getLeft());
  407.             $this->_writeBorderPr($objWriter, 'right',             $pBorders->getRight());
  408.             $this->_writeBorderPr($objWriter, 'top',             $pBorders->getTop());
  409.             $this->_writeBorderPr($objWriter, 'bottom',         $pBorders->getBottom());

  411.         $objWriter->endElement();
  412.     }

  423.     {

  427.             $objWriter->writeAttribute('fontId',             (int)$this->getParentWriter()->getFontHashTable()->getIndexForHashCode($pStyle->getFont()->getHashCode()));
  428.             
  429.             if ($pStyle->getNumberFormat()->getBuiltInFormatCode() === false) {
  430.                 $objWriter->writeAttribute('numFmtId',             (int)($this->getParentWriter()->getNumFmtHashTable()->getIndexForHashCode($pStyle->getNumberFormat()->getHashCode()) + 164)   );
  431.             } else {
  432.                 $objWriter->writeAttribute('numFmtId',             (int)$pStyle->getNumberFormat()->getBuiltInFormatCode());
  433.             }
  434.             
  435.             $objWriter->writeAttribute('fillId',             (int)$this->getParentWriter()->getFillHashTable()->getIndexForHashCode($pStyle->getFill()->getHashCode()));
  436.             $objWriter->writeAttribute('borderId',             (int)$this->getParentWriter()->getBordersHashTable()->getIndexForHashCode($pStyle->getBorders()->getHashCode()));
