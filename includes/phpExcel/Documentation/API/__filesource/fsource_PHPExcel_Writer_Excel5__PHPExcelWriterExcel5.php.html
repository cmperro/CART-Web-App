  214.         // Write the worksheet streams before the global workbook stream,

  221.

  224.  
  225.         // add binary data for sheet streams
  226.         for ($i = 0; $i < $countSheets; ++$i) {

  228.                 $OLE->append($tmp);
  229.             }
  230.         }
  231.  
  232.         $root = new PHPExcel_Shared_OLE_PPS_Root(time(), time(), array($OLE));

  234.             $root->setTempDir($this->_tempDir);
  235.         }

  238.         $res = $root->save($pFilename);
  239.
