  244.     /**
  245.      * Calculate an (approximate) OpenXML column width, based on font size and text contained
  246.      *
  247.      * @param     int        $fontSize            Font size (in pixels or points)
  248.      * @param     bool    $fontSizeInPixels    Is the font size specified in pixels (true) or in points (false) ?
  249.      * @param     string    $columnText            Text to calculate width
  250.      * @param     int        $rotation            Rotation angle
  251.      * @return     int        Column width
  252.      */
  253.     public static function calculateColumnWidth(PHPExcel_Style_Font $font, $columnText = '', $rotation = 0, PHPExcel_Style_Font $defaultFont = null) {
  254.  
  255.         // If it is rich text, use plain text
  256.         if ($columnText instanceof PHPExcel_RichText) {
  257.             $columnText = $columnText->getPlainText();
  258.         }
  259.  
  260.         // Only measure the part before the first newline character (is always "\n")
  261.         if (strpos($columnText, "\n") !== false) {
  262.             $columnText = substr($columnText, 0, strpos($columnText, "\n"));
  263.         }
  264.  
  265.         // Try to get the exact text width in pixels
  266.         try {
  267.             // If autosize method is set to 'approx', use approximation
  268.             if (self::$autoSizeMethod == self::AUTOSIZE_METHOD_APPROX) {
  269.                 throw new Exception('AutoSize method is set to approx');
  270.             }
  271.  
  272.             // Width of text in pixels excl. padding
  273.             $columnWidth = self::getTextWidthPixelsExact($columnText, $font, $rotation);
  274.  
  275.             // Excel adds some padding, use 1.07 of the width of an 'n' glyph
  276.             $columnWidth += ceil(self::getTextWidthPixelsExact('0', $font, 0) * 1.07); // pixels incl. padding
  277.  
  278.         } catch (Exception $e) {
  279.             // Width of text in pixels excl. padding, approximation
  280.             $columnWidth = self::getTextWidthPixelsApprox($columnText, $font, $rotation);
  281.  
  282.             // Excel adds some padding, just use approx width of 'n' glyph
  283.             $columnWidth += self::getTextWidthPixelsApprox('n', $font, 0);
  284.         }
  285.  
  286.         // Convert from pixel width to column width
  287.         $columnWidth = PHPExcel_Shared_Drawing::pixelsToCellDimension($columnWidth, $defaultFont);
  288.  
  289.         // Return
  290.         return round($columnWidth, 6);
  291.     }
  292.  
  293.     /**
  294.      * Get GD text width in pixels for a string of text in a certain font at a certain rotation angle
  295.      *
  296.      * @param string $text 
  297.      * @param PHPExcel_Style_Font 
  298.      * @param int $rotation 
  299.      * @return int 
  300.      * @throws Exception
  301.      */
  302.     public static function getTextWidthPixelsExact($text, PHPExcel_Style_Font $font, $rotation = 0) {
  303.         if (!function_exists('imagettfbbox')) {
  304.             throw new Exception('GD library needs to be enabled');
  305.         }
  306.  
  307.         // font size should really be supplied in pixels in GD2,
  308.         // but since GD2 seems to assume 72dpi, pixels and points are the same
  309.         $fontFile = self::getTrueTypeFontFileFromFont($font);
  310.         $textBox = imagettfbbox($font->getSize(), $rotation, $fontFile, $text);
  311.  
  312.         // Get corners positions
  313.         $lowerLeftCornerX  = $textBox[0];
  314.         $lowerLeftCornerY  = $textBox[1];
  315.         $lowerRightCornerX = $textBox[2];
  316.         $lowerRightCornerY = $textBox[3];
  317.         $upperRightCornerX = $textBox[4];
  318.         $upperRightCornerY = $textBox[5];
  319.         $upperLeftCornerX  = $textBox[6];
  320.         $upperLeftCornerY  = $textBox[7];
  321.         
  322.         // Consider the rotation when calculating the width
  323.         $textWidth = max($lowerRightCornerX - $upperLeftCornerX, $upperRightCornerX - $lowerLeftCornerX);
  324.  
  325.         return $textWidth;
  326.     }

  328.     /**
  329.      * Get approximate width in pixels for a string of text in a certain font at a certain rotation angle
  330.      *
  331.      * @param string $columnText 
  332.      * @param PHPExcel_Style_Font $font 
  333.      * @param int $rotation 
  334.      * @return int Text width in pixels (no padding added)
  335.      */
  336.     public static function getTextWidthPixelsApprox($columnText, PHPExcel_Style_Font $font = null, $rotation = 0)
  337.     {
  338.         $fontName = $font->getName();
  339.         $fontSize = $font->getSize();
  340.  
  341.         // Calculate column width in pixels. We assume fixed glyph width. Result varies with font name and size.
  342.         switch ($fontName) {
  343.             case 'Calibri':
  344.                 // value 8.26 was found via interpolation by inspecting real Excel files with Calibri 11 font.
  345.                 $columnWidth = (int) (8.26 * PHPExcel_Shared_String::CountCharacters($columnText));
  346.                 $columnWidth = $columnWidth * $fontSize / 11; // extrapolate from font size
  347.                 break;
  348.  
  349.             case 'Arial':
  350.                 // value 7 was found via interpolation by inspecting real Excel files with Arial 10 font.
  351.                 $columnWidth = (int) (7 * PHPExcel_Shared_String::CountCharacters($columnText));
  352.                 $columnWidth = $columnWidth * $fontSize / 10; // extrapolate from font size
  353.                 break;
  354.  
  355.             case 'Verdana':
  356.                 // value 8 was found via interpolation by inspecting real Excel files with Verdana 10 font.
  357.                 $columnWidth = (int) (8 * PHPExcel_Shared_String::CountCharacters($columnText));
