  245.     private function _readBSE()
  246.     {
  247.         // offset: 0; size: 2; recVer and recInstance
  248.         
  249.         // bit: 4-15; mask: 0xFFF0; recInstance
  250.         $recInstance = (0xFFF0 & $this->_GetInt2d($this->_data, $this->_pos)) >> 4;
  251.  
  252.         $length = $this->_GetInt4d($this->_data, $this->_pos + 4);
  253.         $recordData = substr($this->_data, $this->_pos + 8, $length);
  254.         
  255.         // move stream pointer to next record
  256.         $this->_pos += 8 + $length;
  257.         
  258.         // add BSE to BstoreContainer
  259.         $BSE = new PHPExcel_Shared_Escher_DggContainer_BstoreContainer_BSE();
  260.         $this->_object->addBSE($BSE);
  261.  
  262.         $BSE->setBLIPType($recInstance);
  263.         
  264.         // offset: 0; size: 1; btWin32 (MSOBLIPTYPE)
  265.         $btWin32 = ord($recordData[0]);
  266.  
  267.         // offset: 1; size: 1; btWin32 (MSOBLIPTYPE)
  268.         $btMacOS = ord($recordData[1]);
  269.  
  270.         // offset: 2; size: 16; MD4 digest
  271.         $rgbUid = substr($recordData, 2, 16);
  272.  
  273.         // offset: 18; size: 2; tag
  274.         $tag = $this->_GetInt2d($recordData, 18);
  275.  
  276.         // offset: 20; size: 4; size of BLIP in bytes
  277.         $size = $this->_GetInt4d($recordData, 20);
  278.  
  279.         // offset: 24; size: 4; number of references to this BLIP
  280.         $cRef = $this->_GetInt4d($recordData, 24);
  281.  
  282.         // offset: 28; size: 4; MSOFO file offset

  289.         $cbName = ord($recordData{33});
  290.  
  291.         // offset: 34; size: 1; unused2

  293.  
  294.         // offset: 35; size: 1; unused3
  295.         $unused3 = ord($recordData{35});

  630.      *

  634.     private function _readOfficeArtRGFOPTE($data, $n) {
  635.  
  636.         $splicedComplexData = substr($data, 6 * $n);
  637.  
  638.         // loop through property-value pairs
