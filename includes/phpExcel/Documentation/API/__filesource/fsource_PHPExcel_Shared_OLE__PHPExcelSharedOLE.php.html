  290.             switch ($type) {
  291.             case self::OLE_PPS_TYPE_ROOT:
  292.                 $pps = new PHPExcel_Shared_OLE_PPS_Root(null, null, array());

  294.                 break;
  295.             case self::OLE_PPS_TYPE_DIR:
  296.                 $pps = new PHPExcel_Shared_OLE_PPS(null, null, null, null, null,
  297.                                    null, null, null, null, array());
  298.                 break;
  299.             case self::OLE_PPS_TYPE_FILE:
  300.                 $pps = new PHPExcel_Shared_OLE_PPS_File($name);
  301.                 break;
  302.             default:
  303.                 continue;
  304.             }
  305.             fseek($fh, 1, SEEK_CUR);
  306.             $pps->Type    = $type;
  307.             $pps->Name    = $name;
  308.             $pps->PrevPps = $this->_readInt4($fh);
  309.             $pps->NextPps = $this->_readInt4($fh);
  310.             $pps->DirPps  = $this->_readInt4($fh);
  311.             fseek($fh, 20, SEEK_CUR);

  313.             $pps->Time2nd = self::OLE2LocalDate(fread($fh, 8));
  314.             $pps->_StartBlock = $this->_readInt4($fh);

  321.                 $this->_ppsTreeComplete($this->root->No)) {
  322.  
  323.                 break;

  325.         }

  337.                         $nos[] = $childPps->PrevPps;
  338.                         $nos[] = $childPps->NextPps;
  339.                         $pps->children[] = $childPps;
  340.                     }
  341.                 }
  342.             }
  343.         }

  345.         return true;
