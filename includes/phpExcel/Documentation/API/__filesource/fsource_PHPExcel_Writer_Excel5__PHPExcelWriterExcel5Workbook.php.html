  1338.     /**

  1340.      * additional continuation byte depending on whether the string is split between
  1341.      * blocks or whether it starts at the beginning of the block. (There are also
  1342.      * additional complications that will arise later when/if Rich Strings are
  1343.      * supported).
  1344.      *
  1345.      * The Excel documentation says that the SST record should be followed by an
  1346.      * EXTSST record. The EXTSST record is a hash table that is used to optimise
  1347.      * access to SST. However, despite the documentation it doesn't seem to be
  1348.      * required so we will ignore it.
  1349.      *
  1350.      * @return string Binary data
  1351.      */
  1352.     private function _writeSharedStringsTable()
  1353.     {
  1354.         // maximum size of record data (excluding record header)
  1355.         $continue_limit = 8224;
  1356.  
  1357.         // initialize array of record data blocks
  1358.         $recordDatas = array();
  1359.  
  1360.         // start SST record data block with total number of strings, total number of unique strings
  1361.         $recordData = pack("VV", $this->_str_total, $this->_str_unique);
  1362.  
  1363.         // loop through all (unique) strings in shared strings table
  1364.         foreach (array_keys($this->_str_table) as $string) {
  1365.  
  1366.             // here $string is a BIFF8 encoded string
  1367.  
  1368.             // length = character count
  1369.             $headerinfo = unpack("vlength/Cencoding", $string);
  1370.  
  1371.             // currently, this is always 1 = uncompressed
  1372.             $encoding = $headerinfo["encoding"];
  1373.  
  1374.             // initialize finished writing current $string
  1375.             $finished = false;
  1376.  
  1377.             while ($finished === false) {
  1378.  
  1379.                 // normally, there will be only one cycle, but if string cannot immediately be written as is
  1380.                 // there will be need for more than one cylcle, if string longer than one record data block, there
  1381.                 // may be need for even more cycles
  1382.  
  1383.                 if (strlen($recordData) + strlen($string) < $continue_limit) {
  1384.                     // then we can write the string (or remainder of string) without any problems
  1385.                     $recordData .= $string;
  1386.  
  1387.                     // we are finished writing this string
  1388.                     $finished = true;
  1389.  
  1390.                 } else if (strlen($recordData) + strlen($string) == $continue_limit) {
  1391.                     // then we can also write the string (or remainder of string)
  1392.                     $recordData .= $string;
  1393.  
  1394.                     // but we close the record data block, and initialize a new one
  1395.                     $recordDatas[] = $recordData;
  1396.                     $recordData = '';
  1397.  
  1398.                     // we are finished writing this string
  1399.                     $finished = true;
  1400.  
  1401.                 } else {
  1402.                     // special treatment writing the string (or remainder of the string)
  1403.                     // If the string is very long it may need to be written in more than one CONTINUE record.
  1404.  
  1405.                     // check how many bytes more there is room for in the current record
  1406.                     $space_remaining = $continue_limit - strlen($recordData);
  1407.  
  1408.                     // minimum space needed
  1409.                     // uncompressed: 2 byte string length length field + 1 byte option flags + 2 byte character
  1410.                     // compressed:   2 byte string length length field + 1 byte option flags + 1 byte character
  1411.                     $min_space_needed = ($encoding == 1) ? 5 : 4;
  1412.  
  1413.                     // We have two cases
  1414.                     // 1. space remaining is less than minimum space needed
  1415.                     //        here we must waste the space remaining and move to next record data block
  1416.                     // 2. space remaining is greater than or equal to minimum space needed
  1417.                     //        here we write as much as we can in the current block, then move to next record data block
  1418.  
  1419.                     // 1. space remaining is less than minimum space needed
  1420.                     if ($space_remaining < $min_space_needed) {
  1421.                         // we close the block, store the block data
  1422.                         $recordDatas[] = $recordData;
  1423.  
  1424.                         // and start new record data block where we start writing the string
  1425.                         $recordData = '';
  1426.  
  1427.                     // 2. space remaining is greater than or equal to minimum space needed
  1428.                     } else {
  1429.                         // initialize effective remaining space, for Unicode strings this may need to be reduced by 1, see below

  1431.  
  1432.                         // for uncompressed strings, sometimes effective space remaining is reduced by 1
  1433.                         if ( $encoding == 1 && (strlen($string) - $space_remaining) % 2 == 1 ) {
  1434.                             --$effective_space_remaining;
  1435.                         }
  1436.  
  1437.                         // one block fininshed, store the block data
  1438.                         $recordData .= substr($string, 0, $effective_space_remaining);
  1439.  
  1440.                         $string = substr($string, $effective_space_remaining); // for next cycle in while loop
  1441.                         $recordDatas[] = $recordData;
  1442.  
  1443.                         // start new record data block with the repeated option flags
  1444.                         $recordData = pack('C', $encoding);
  1445.                     }
  1446.                 }
  1447.             }
  1448.         }
  1449.  
  1450.         // Store the last record data block unless it is empty
  1451.         // if there was no need for any continue records, this will be the for SST record data block itself
  1452.         if (strlen($recordData) > 0) {
  1453.             $recordDatas[] = $recordData;
  1454.         }
  1455.  
  1456.         // combine into one chunk with all the blocks SST, CONTINUE,...
  1457.         $chunk = '';
  1458.         foreach ($recordDatas as $i => $recordData) {

  1461.  
  1462.             $header = pack("vv", $record, strlen($recordData));
  1463.             $data = $header . $recordData;
  1464.  
  1465.             $chunk .= $this->writeData($data);
  1466.         }
  1467.  
  1468.         return $chunk;
  1469.     }
