  54.     /**
  55.      * Get the width of a column in pixels. We use the relationship y = ceil(7x) where
  56.      * x is the width in intrinsic Excel units (measuring width in number of normal characters)
  57.      * This holds for Arial 10
  58.      *
  59.      * @param PHPExcel_Worksheet $sheet The sheet
  60.      * @param integer $col The column
  61.      * @return integer The width in pixels
  62.     */
  63.     public static function sizeCol($sheet, $col = 'A')
  64.     {
  65.         // default font of the workbook
  66.         $font = $sheet->getParent()->getDefaultStyle()->getFont();
  67.  
  68.         $columnDimensions = $sheet->getColumnDimensions();
  69.  
  70.         // first find the true column width in pixels (uncollapsed and unhidden)
  71.         if ( isset($columnDimensions[$col]) and $columnDimensions[$col]->getWidth() != -1 ) {
  72.  
  73.             // then we have column dimension with explicit width
  74.             $columnDimension = $columnDimensions[$col];
  75.             $width = $columnDimension->getWidth();
  76.             $pixelWidth = PHPExcel_Shared_Drawing::cellDimensionToPixels($width, $font);
  77.  
  78.         } else if ($sheet->getDefaultColumnDimension()->getWidth() != -1) {
  79.  
  80.             // then we have default column dimension with explicit width
  81.             $defaultColumnDimension = $sheet->getDefaultColumnDimension();
  82.             $width = $defaultColumnDimension->getWidth();
  83.             $pixelWidth = PHPExcel_Shared_Drawing::cellDimensionToPixels($width, $font);
  84.  
  85.         } else {
  86.  
  87.             // we don't even have any default column dimension. Width depends on default font
  88.             $pixelWidth = PHPExcel_Shared_Font::getDefaultColumnWidthByFont($font, true);
  89.         }
  90.  
  91.         // now find the effective column width in pixels
  92.         if (isset($columnDimensions[$col]) and !$columnDimensions[$col]->getVisible()) {
  93.             $effectivePixelWidth = 0;
  94.         } else {
  95.             $effectivePixelWidth = $pixelWidth;
  96.         }
  97.  
  98.         return $effectivePixelWidth;
  99.     }

  101.     /**
  102.      * Convert the height of a cell from user's units to pixels. By interpolation
  103.      * the relationship is: y = 4/3x. If the height hasn't been set by the user we
  104.      * use the default value. If the row is hidden we use a value of zero.
  105.      *
  106.      * @param PHPExcel_Worksheet $sheet The sheet
  107.      * @param integer $row The row index (1-based)
  108.      * @return integer The width in pixels
  109.      */
  110.     public static function sizeRow($sheet, $row = 1)
  111.     {
  112.         // default font of the workbook
  113.         $font = $sheet->getParent()->getDefaultStyle()->getFont();
  114.  
  115.         $rowDimensions = $sheet->getRowDimensions();
  116.  
  117.         // first find the true row height in pixels (uncollapsed and unhidden)
  118.         if ( isset($rowDimensions[$row]) and $rowDimensions[$row]->getRowHeight() != -1) {
  119.  
  120.             // then we have a row dimension
  121.             $rowDimension = $rowDimensions[$row];
  122.             $rowHeight = $rowDimension->getRowHeight();
  123.             $pixelRowHeight = (int) ceil(4 * $rowHeight / 3); // here we assume Arial 10
  124.  
  125.         } else if ($sheet->getDefaultRowDimension()->getRowHeight() != -1) {
  126.  
  127.             // then we have a default row dimension with explicit height
  128.             $defaultRowDimension = $sheet->getDefaultRowDimension();
  129.             $rowHeight = $defaultRowDimension->getRowHeight();
  130.             $pixelRowHeight = PHPExcel_Shared_Drawing::pointsToPixels($rowHeight);
  131.  
  132.         } else {
  133.  
  134.             // we don't even have any default row dimension. Height depends on default font
  135.             $pointRowHeight = PHPExcel_Shared_Font::getDefaultRowHeightByFont($font);
  136.             $pixelRowHeight = PHPExcel_Shared_Font::fontSizeToPixels($pointRowHeight);
  137.  
  138.         }
  139.  
  140.         // now find the effective row height in pixels
  141.         if ( isset($rowDimensions[$row]) and !$rowDimensions[$row]->getVisible() ) {
  142.             $effectivePixelRowHeight = 0;
  143.         } else {
  144.             $effectivePixelRowHeight = $pixelRowHeight;
  145.         }
  146.  
  147.         return $effectivePixelRowHeight;
  148.     }

  150.     /**
  151.      * Get the horizontal distance in pixels between two anchors
  152.      * The distanceX is found as sum of all the spanning columns widths minus correction for the two offsets
  153.      *
  154.      * @param PHPExcel_Worksheet $sheet 
  155.      * @param string $startColumn 
  156.      * @param integer $startOffset Offset within start cell measured in 1/1024 of the cell width
  157.      * @param string $endColumn 
  158.      * @param integer $endOffset Offset within end cell measured in 1/1024 of the cell width
  159.      * @return integer Horizontal measured in pixels
  160.      */
  161.     public static function getDistanceX(PHPExcel_Worksheet $sheet, $startColumn = 'A', $startOffsetX = 0, $endColumn = 'A', $endOffsetX = 0)
  162.     {
  163.         $distanceX = 0;
  164.  
  165.         // add the widths of the spanning columns
  166.         $startColumnIndex = PHPExcel_Cell::columnIndexFromString($startColumn) - 1; // 1-based
  167.         $endColumnIndex = PHPExcel_Cell::columnIndexFromString($endColumn) - 1; // 1-based
  168.         for ($i = $startColumnIndex; $i <= $endColumnIndex; ++$i) {
  169.             $distanceX += self::sizeCol($sheet, PHPExcel_Cell::stringFromColumnIndex($i));
  170.         }
  171.  
  172.         // correct for offsetX in startcell
  173.         $distanceX -= (int) floor(self::sizeCol($sheet, $startColumn) * $startOffsetX / 1024);
  174.  
  175.         // correct for offsetX in endcell
  176.         $distanceX -= (int) floor(self::sizeCol($sheet, $endColumn) * (1 - $endOffsetX / 1024));
  177.  
  178.         return $distanceX;
  179.     }

  181.     /**
  182.      * Get the vertical distance in pixels between two anchors
  183.      * The distanceY is found as sum of all the spanning rows minus two offsets
  184.      *
  185.      * @param PHPExcel_Worksheet $sheet 
  186.      * @param string $startRow (1-based)
  187.      * @param integer $startOffset Offset within start cell measured in 1/256 of the cell height
  188.      * @param string $endRow (1-based)
  189.      * @param integer $endOffset Offset within end cell measured in 1/256 of the cell height
  190.      * @return integer Vertical distance measured in pixels
  191.      */
  192.     public static function getDistanceY(PHPExcel_Worksheet $sheet, $startRow = 1, $startOffsetY = 0, $endRow = 1, $endOffsetY = 0)
  193.     {
  194.         $distanceY = 0;
  195.  
  196.         // add the widths of the spanning rows
  197.         for ($row = $startRow; $row <= $endRow; ++$row) {
  198.             $distanceY += self::sizeRow($sheet, $row);
  199.         }
  200.  
  201.         // correct for offsetX in startcell
  202.         $distanceY -= (int) floor(self::sizeRow($sheet, $startRow) * $startOffsetY / 256);
  203.  
  204.         // correct for offsetX in endcell
  205.         $distanceY -= (int) floor(self::sizeRow($sheet, $endRow) * (1 - $endOffsetY / 256));
  206.  
  207.         return $distanceY;
  208.     }

  210.     /**
  211.      * Convert 1-cell anchor coordinates to 2-cell anchor coordinates
  212.      * This function is ported from PEAR Spreadsheet_Writer_Excel with small modifications
  213.      *
  214.      * Calculate the vertices that define the position of the image as required by
  215.      * the OBJ record.
  216.      *
  217.      *         +------------+------------+
  218.      *         |     A      |      B     |
  219.      *   +-----+------------+------------+
  220.      *   |     |(x1,y1)     |            |
  221.      *   |  1  |(A1)._______|______      |
  222.      *   |     |    |              |     |
  223.      *   |     |    |              |     |
  224.      *   +-----+----|    BITMAP    |-----+
  225.      *   |     |    |              |     |
  226.      *   |  2  |    |______________.     |
  227.      *   |     |            |        (B2)|
  228.      *   |     |            |     (x2,y2)|
  229.      *   +---- +------------+------------+
  230.      *
  231.      * Example of a bitmap that covers some of the area from cell A1 to cell B2.
  232.      *
  233.      * Based on the width and height of the bitmap we need to calculate 8 vars:
  234.      *     $col_start, $row_start, $col_end, $row_end, $x1, $y1, $x2, $y2.
  235.      * The width and height of the cells are also variable and have to be taken into
  236.      * account.
  237.      * The values of $col_start and $row_start are passed in from the calling
  238.      * function. The values of $col_end and $row_end are calculated by subtracting
  239.      * the width and height of the bitmap from the width and height of the
  240.      * underlying cells.
  241.      * The vertices are expressed as a percentage of the underlying cell width as
  242.      * follows (rhs values are in pixels):
  243.      *
  244.      *       x1 = X / W *1024
  245.      *       y1 = Y / H *256
  246.      *       x2 = (X-1) / W *1024
  247.      *       y2 = (Y-1) / H *256
  248.      *
  249.      *       Where:  X is distance from the left side of the underlying cell
  250.      *               Y is distance from the top of the underlying cell
  251.      *               W is the width of the cell
  252.      *               H is the height of the cell
  253.      *
  254.      * @param PHPExcel_Worksheet $sheet 
  255.      * @param string $coordinates E.g. 'A1'
  256.      * @param integer $offsetX Horizontal offset in pixels
  257.      * @param integer $offsetY Vertical offset in pixels
  258.      * @param integer $width Width in pixels
  259.      * @param integer $height Height in pixels
  260.      * @return array 
  261.      */
  262.     public static function oneAnchor2twoAnchor($sheet, $coordinates, $offsetX, $offsetY, $width, $height)
  263.     {
  264.         list($column, $row) = PHPExcel_Cell::coordinateFromString($coordinates);
  265.         $col_start = PHPExcel_Cell::columnIndexFromString($column) - 1;
  266.         $row_start = $row - 1;
  267.         
  268.         $x1 = $offsetX;
  269.         $y1 = $offsetY;
  270.  
  271.         // Initialise end cell to the same as the start cell
  272.         $col_end    = $col_start;  // Col containing lower right corner of object
  273.         $row_end    = $row_start;  // Row containing bottom right corner of object
  274.  
  275.         // Zero the specified offset if greater than the cell dimensions
  276.         if ($x1 >= self::sizeCol($sheet, PHPExcel_Cell::stringFromColumnIndex($col_start))) {
  277.             $x1 = 0;
  278.         }
  279.         if ($y1 >= self::sizeRow($sheet, $row_start + 1)) {
  280.             $y1 = 0;
  281.         }
  282.  
  283.         $width      = $width  + $x1 -1;
  284.         $height     = $height + $y1 -1;
  285.  
  286.         // Subtract the underlying cell widths to find the end cell of the image
  287.         while ($width >= self::sizeCol($sheet, PHPExcel_Cell::stringFromColumnIndex($col_end))) {
  288.             $width -= self::sizeCol($sheet, PHPExcel_Cell::stringFromColumnIndex($col_end));
  289.             ++$col_end;
  290.         }
  291.  
  292.         // Subtract the underlying cell heights to find the end cell of the image
  293.         while ($height >= self::sizeRow($sheet, $row_end + 1)) {
  294.             $height -= self::sizeRow($sheet, $row_end + 1);
  295.             ++$row_end;
  296.         }
  297.  
  298.         // Bitmap isn't allowed to start or finish in a hidden cell, i.e. a cell
  299.         // with zero height or width.
  300.         if (self::sizeCol($sheet, PHPExcel_Cell::stringFromColumnIndex($col_start)) == 0) {
  301.             return;
  302.         }
  303.         if (self::sizeCol($sheet, PHPExcel_Cell::stringFromColumnIndex($col_end))   == 0) {
  304.             return;
  305.         }
  306.         if (self::sizeRow($sheet, $row_start + 1) == 0) {
  307.             return;
  308.         }
  309.         if (self::sizeRow($sheet, $row_end + 1)   == 0) {
  310.             return;
  311.         }
  312.  
  313.         // Convert the pixel values to the percentage value expected by Excel
  314.         $x1 = $x1     / self::sizeCol($sheet, PHPExcel_Cell::stringFromColumnIndex($col_start))   * 1024;
  315.         $y1 = $y1     / self::sizeRow($sheet, $row_start + 1)   *  256;
  316.         $x2 = ($width + 1)  / self::sizeCol($sheet, PHPExcel_Cell::stringFromColumnIndex($col_end))     * 1024; // Distance to right side of object
  317.         $y2 = ($height + 1) / self::sizeRow($sheet, $row_end + 1)     *  256; // Distance to bottom of object
  318.  
  319.         $startCoordinates = PHPExcel_Cell::stringFromColumnIndex($col_start) . ($row_start + 1);
  320.         $endCoordinates = PHPExcel_Cell::stringFromColumnIndex($col_end) . ($row_end + 1);
  321.  
  322.         $twoAnchor = array(
  323.             'startCoordinates' => $startCoordinates,
  324.             'startOffsetX' => $x1,
  325.             'startOffsetY' => $y1,
  326.             'endCoordinates' => $endCoordinates,
  327.             'endOffsetX' => $x2,
  328.             'endOffsetY' => $y2,
  329.         );
  330.  
  331.         return  $twoAnchor;
  332.     }
